## operadores:

# Aritméticos:

    5 + 2 = 7 
    5 - 2 = 3
    5 * 2 = 10
    5 / 2 = 2.5
    5 % 2 = 1 --> resto da divisão
    5 ** 2 = 25

    Ordem de precedência
    () -> ** -> * / % -> + -

# Atribuição:
    '''
    var a = 1
    '''
# Relacionais:

    5 < 2 = False
    7 > 4 = True
    8 >= 8 = True
    7 <= 9 = True
    8 == 8 = True
    4 != 4 = False

# Lógicos:

    ! = Negação
    && = e conjunção
    || = ou

    Ordem de precedência
    ! -> && -> ||

# Ternário:

    função ? se true : se false //(iif)

    teste ? true: falso
